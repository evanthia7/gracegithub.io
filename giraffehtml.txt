        // Conversion factors
        const conversionFactors = {
            meter: {
                foot: 3.28084,
                inch: 39.3701,
                cm: 100,
                km: 0.001,
                mile: 0.000621371
            },
            foot: {
                meter: 0.3048,
                inch: 12,
                cm: 30.48,
                km: 0.0003048,
                mile: 0.000189394
            },
            inch: {
                meter: 0.0254,
                foot: 0.0833333,
                cm: 2.54,
                km: 0.0000254,
                mile: 0.000015783
            },
            cm: {
                meter: 0.01,
                foot: 0.0328084,
                inch: 0.393701,
                km: 0.00001,
                mile: 0.00000621371
            },
            km: {
                meter: 1000,
                foot: 3280.84,
                inch: 39370.1,
                cm: 100000,
                mile: 0.621371
            },
            mile: {
                meter: 1609.34,
                foot: 5280,
                inch: 63360,
                cm: 160934,
                km: 1.60934
            }
        };
        
        // DOM Elements
        const valueInput = document.getElementById('value');
        const fromUnitSelect = document.getElementById('from-unit');
        const toUnitSelect = document.getElementById('to-unit');
        const convertBtn = document.getElementById('convert-btn');
        const swapBtn = document.getElementById('swap-btn');
        const resultDiv = document.getElementById('result');
        
        // Convert function
        function convert() {
            const value = parseFloat(valueInput.value);
            
            if (isNaN(value)) {
                alert('Please enter a valid number');
                return;
            }
            
            const fromUnit = fromUnitSelect.value;
            const toUnit = toUnitSelect.value;
            
            if (fromUnit === toUnit) {
                resultDiv.textContent = `${value} ${fromUnit} = ${value} ${toUnit}`;
                resultDiv.style.display = 'block';
                return;
            }
            
            let result;
            
            if (conversionFactors[fromUnit] && conversionFactors[fromUnit][toUnit]) {
                result = value * conversionFactors[fromUnit][toUnit];
            } else if (conversionFactors[toUnit] && conversionFactors[toUnit][fromUnit]) {
                result = value / conversionFactors[toUnit][fromUnit];
            } else {
                alert('Conversion not supported');
                return;
            }
            
            // Format numbers
            const formattedValue = value % 1 === 0 ? value : value.toFixed(2);
            const formattedResult = result % 1 === 0 ? result : result.toFixed(4);
            
            // Get display names
            const fromDisplay = fromUnitSelect.options[fromUnitSelect.selectedIndex].text;
            const toDisplay = toUnitSelect.options[toUnitSelect.selectedIndex].text;
            
            resultDiv.textContent = `${formattedValue} ${fromDisplay} = ${formattedResult} ${toDisplay}`;
            resultDiv.style.display = 'block';
        }
        
        // Swap units function
        function swapUnits() {
            const tempUnit = fromUnitSelect.value;
            fromUnitSelect.value = toUnitSelect.value;
            toUnitSelect.value = tempUnit;
            convert();
        }
        
        // Event listeners
        convertBtn.addEventListener('click', convert);
        swapBtn.addEventListener('click', swapUnits);
        
        // Initialize with giraffe conversion
        window.addEventListener('DOMContentLoaded', convert);
    </script>
